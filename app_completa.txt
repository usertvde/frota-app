    (function() {
      const supabase = window.supabase;
      if (!supabase) { document.body.innerHTML = '<h1 style="color:red;text-align:center;padding:20px;">Erro: Sistema não inicializado. Contacte o administrador.</h1>'; return; }

      let currentReportData = [];
      let allVehiclesData = [];

      console.log('admin.html carregado com sucesso');

      if ('serviceWorker' in navigator) {
        navigator.serviceWorker.register('/sw.js');
      }

      document.getElementById('logout-btn').addEventListener('click', async () => {
        await supabase.auth.signOut();
        window.location.href = 'index.html';
      });

      document.querySelectorAll('.tab').forEach(tab => {
        tab.addEventListener('click', () => {
          document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
          document.querySelectorAll('.panel').forEach(p => p.classList.remove('active'));
          tab.classList.add('active');
          const panelId = tab.dataset.tab + '-panel';
          document.getElementById(panelId).classList.add('active');
          if (tab.dataset.tab === 'shifts') loadShiftsPanel();
          if (tab.dataset.tab === 'reports') loadReportDropdowns();
        });
      });

      (async () => {
        try {
          const { data: { session }, error } = await supabase.auth.getSession();
          if (error || !session) { window.location.href = 'index.html'; return; }
          sessionToken = session.access_token;

          const { data: profile } = await supabase.from('profiles').select('role').eq('user_id', session.user.id).single();
          if (!profile || profile.role !== 'admin') { window.location.href = 'index.html'; return; }

          console.log('Admin autenticado');
          setupForms();
          loadVehicles();
          loadDrivers();
          loadShiftsPanel();
          loadAdminSettings();
        } catch (err) {
          console.error('Erro inicial:', err);
        }
      })();

        try {
          const { data, error } = await supabase.from('vehicles').select('*').order('id');
          if (error) throw error;
          allVehiclesData = data || [];
          const container = document.getElementById('vehicles-list');
          if (!data || data.length === 0) {
            container.innerHTML = '<p class="empty-msg">Nenhum veículo registado.</p>';
            return;
          }
          container.innerHTML = data.map(v => `
            <div class="card" id="vehicle-${v.id}">
              <div class="vehicle-display">
                ${v.photo_url ? `<img src="${v.photo_url}" alt="Foto veículo" class="vehicle-img">` : '<div class="no-photo">📷 Sem foto</div>'}
                <div class="vehicle-info">
                  <strong>${v.brand} ${v.model}</strong>
                  <span>Ano: ${v.year || 'N/A'} | Matrícula: ${v.license_plate}</span>
                </div>
                <div class="card-actions">
                  <button class="edit-btn" data-edit-vehicle="${v.id}">✏️ Editar</button>
                  <button class="delete-btn" data-delete-vehicle="${v.id}">🗑️ Apagar</button>
                </div>
              </div>
              <div class="vehicle-edit" style="display:none;">
                <input type="text" class="edit-brand" value="${v.brand}" placeholder="Marca">
                <input type="text" class="edit-model" value="${v.model}" placeholder="Modelo">
                <input type="text" class="edit-plate" value="${v.license_plate}" placeholder="Matrícula">

      function setupVehicleListeners() {
        document.querySelectorAll('[data-edit-vehicle]').forEach(btn => {
          btn.addEventListener('click', () => {
            const id = btn.dataset.editVehicle;
            document.querySelector(`#vehicle-${id} .vehicle-display`).style.display = 'none';
            document.querySelector(`#vehicle-${id} .vehicle-edit`).style.display = 'block';
          });
        });
        document.querySelectorAll('[data-cancel-vehicle]').forEach(btn => {
          btn.addEventListener('click', () => {
            const id = btn.dataset.cancelVehicle;
            document.querySelector(`#vehicle-${id} .vehicle-display`).style.display = 'flex';
            document.querySelector(`#vehicle-${id} .vehicle-edit`).style.display = 'none';
          });
        });
        document.querySelectorAll('[data-save-vehicle]').forEach(btn => {
          btn.addEventListener('click', async () => {
            const id = btn.dataset.saveVehicle;
            const brand = document.querySelector(`#vehicle-${id} .edit-brand`).value;
            const model = document.querySelector(`#vehicle-${id} .edit-model`).value;
            const plate = document.querySelector(`#vehicle-${id} .edit-plate`).value;
            const year = document.querySelector(`#vehicle-${id} .edit-year`).value;
            await supabase.from('vehicles').update({ brand, model, license_plate: plate, year: parseInt(year) || null }).eq('id', id);
            loadVehicles();
          });
        });
        document.querySelectorAll('[data-delete-vehicle]').forEach(btn => {
          btn.addEventListener('click', async () => {
            if (confirm('Apagar este veículo?')) {
              await supabase.from('vehicles').delete().eq('id', btn.dataset.deleteVehicle);
              loadVehicles();
              loadShiftsPanel();
            }
          });
        });
      }

      function setupDriverListeners() {
        document.querySelectorAll('[data-edit-driver]').forEach(btn => {
          btn.addEventListener('click', () => {
            const id = btn.dataset.editDriver;
            document.querySelector(`#driver-${id} .driver-display`).style.display = 'none';
            document.querySelector(`#driver-${id} .driver-edit`).style.display = 'block';
          });
        });
        document.querySelectorAll('[data-cancel-driver]').forEach(btn => {
          btn.addEventListener('click', () => {
            const id = btn.dataset.cancelDriver;
            document.querySelector(`#driver-${id} .driver-display`).style.display = 'flex';
            document.querySelector(`#driver-${id} .driver-edit`).style.display = 'none';
          });
        });
        document.querySelectorAll('[data-save-driver]').forEach(btn => {
          btn.addEventListener('click', async () => {
            const id = btn.dataset.saveDriver;
            const fullName = document.querySelector(`#driver-${id} .edit-fullname`).value;
            const accessNumber = document.querySelector(`#driver-${id} .edit-access`).value;
            await supabase.from('profiles').update({ full_name: fullName, access_number: accessNumber }).eq('user_id', id);
            loadDrivers();
            loadShiftsPanel();
          });
        });
        document.querySelectorAll('[data-reset]').forEach(btn => {
          btn.addEventListener('click', async () => {
            const res = await fetch(window.EDGE_FUNCTION_URL, {
              method: 'POST',
              headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${sessionToken}` },
              body: JSON.stringify({ action: 'reset-password', email: btn.dataset.reset })
            });
            const result = await res.json();
            alert(result.error ? 'Erro: ' + result.error : 'Email de recuperação enviado para ' + btn.dataset.reset);
          });
        });
        document.querySelectorAll('[data-delete]').forEach(btn => {
          btn.addEventListener('click', async () => {
            if (!confirm('Apagar este colaborador? Esta ação é irreversível.')) return;
            const res = await fetch(window.EDGE_FUNCTION_URL, {
              method: 'POST',
              headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${sessionToken}` },
              body: JSON.stringify({ action: 'delete-driver', userId: btn.dataset.delete })
            });
            await res.json();
            loadDrivers();
            loadShiftsPanel();
          });
        });
      }

      // ---------- HORÁRIOS ----------
      async function loadShiftsPanel() {
        try {
          const { data: drivers } = await supabase.from('profiles').select('user_id, full_name').eq('role', 'driver');
          const driverSelect = document.getElementById('admin-driver-select');
          driverSelect.innerHTML = '<option value="">Selecione um colaborador</option>' + (drivers || []).map(d => `<option value="${d.user_id}">${d.full_name}</option>`).join('');

          const { data: vehicles } = await supabase.from('vehicles').select('id, brand, model, license_plate');
          allVehiclesData = vehicles || [];
          const vehicleSelect = document.getElementById('admin-shift-vehicle');
          vehicleSelect.innerHTML = '<option value="">Selecione um veículo</option>' + allVehiclesData.map(v => `<option value="${v.id}">${v.brand} ${v.model} (${v.license_plate})</option>`).join('');

          const { data: shifts } = await supabase
            .from('shifts')
            .select('*, vehicles(license_plate, brand, model), profiles!driver_id(full_name)')
            .order('date');
          allExistingShifts = shifts || [];

          const shiftsContainer = document.getElementById('shifts-list');
          if (!shifts || shifts.length === 0) {
            shiftsContainer.innerHTML = '<p class="empty-msg">Nenhum horário criado.</p>';
          } else {
            shiftsContainer.innerHTML = shifts.map(s => `
              <div class="card">
                <strong>👤 ${s.profiles?.full_name || 'N/A'}</strong> — 🚗 ${s.vehicles?.brand || ''} ${s.vehicles?.model || ''} (${s.vehicles?.license_plate || 'N/A'})<br>
                📅 ${s.date} 🕐 ${s.start_time?.substring(0,5)}-${s.end_time?.substring(0,5)} | <span class="status-badge status-${s.status}">${s.status}</span>
                <button class="delete-btn-sm" data-delete-shift="${s.id}">🗑️ Remover</button>
              </div>
            `).join('');

            // Event listeners para remover horário (AGORA FUNCIONA com ON DELETE CASCADE)
            shiftsContainer.querySelectorAll('[data-delete-shift]').forEach(btn => {
              btn.addEventListener('click', async () => {
                if (confirm('Remover este horário? O relatório associado também será apagado.')) {
                  const { error } = await supabase.from('shifts').delete().eq('id', btn.dataset.deleteShift);
                  if (error) {
                    alert('Erro ao remover: ' + error.message);
                  } else {
                    loadShiftsPanel();
                  }
                }
              });
            });
          }

          // Resetar formulário
          document.getElementById('admin-shift-date').value = '';
          document.getElementById('admin-shift-start').value = '';
          document.getElementById('admin-shift-end').value = '';
          document.getElementById('admin-shift-date-display').textContent = '-';
          document.getElementById('admin-shift-time-display').textContent = '-';
          document.getElementById('vehicle-conflict-warning').style.display = 'none';
          document.getElementById('create-shift-btn').disabled = true;
          document.getElementById('admin-availability-grid').innerHTML = '<p>Selecione um colaborador para ver as disponibilidades.</p>';

        } catch (err) {
          console.error('Erro ao carregar painel de horários:', err);
        }
      }

      // Evento: selecionar colaborador para ver disponibilidades
      document.getElementById('admin-driver-select').addEventListener('change', async function() {
        const driverId = this.value;
        const gridContainer = document.getElementById('admin-availability-grid');

        if (!driverId) {
          gridContainer.innerHTML = '<p>Selecione um colaborador para ver as disponibilidades.</p>';
          return;
        }

        try {
          const { data: avail } = await supabase
            .from('driver_availability')
            .select('*')
            .eq('driver_id', driverId)
            .gte('date', new Date().toISOString().split('T')[0])
            .order('date');

          const availabilityData = avail || [];

          if (availabilityData.length === 0) {
            gridContainer.innerHTML = '<p class="empty-msg">Este colaborador não tem disponibilidade registada.</p>';
            return;
          }

          let html = '<table class="availability-table"><thead><tr><th>Data</th><th>Horário</th><th>Ação</th></tr></thead><tbody>';

          availabilityData.forEach(a => {
            const dateLabel = a.date.split('-').reverse().join('/');
            const timeLabel = a.start_time?.substring(0,5) + ' - ' + a.end_time?.substring(0,5);

            const existingShift = allExistingShifts.find(s =>
              s.driver_id === driverId &&
              s.date === a.date &&
              s.start_time === a.start_time &&
              s.end_time === a.end_time
            );

            if (existingShift) {
              html += `<tr style="background:#d4edda;">
                <td>${dateLabel}</td>
                <td>${timeLabel}</td>
                <td>✅ ${existingShift.vehicles?.license_plate || 'Atribuído'}</td>
              </tr>`;
            } else {
              html += `<tr class="avail-row" data-date="${a.date}" data-start="${a.start_time}" data-end="${a.end_time}" style="cursor:pointer;">
                <td>${dateLabel}</td>
                <td>${timeLabel}</td>
                <td><button class="select-avail-btn" style="font-size:12px;padding:5px 10px;">Selecionar</button></td>
              </tr>`;
            }
          });

          html += '</tbody></table>';
          gridContainer.innerHTML = html;

          // Event listeners para selecionar disponibilidade
          gridContainer.querySelectorAll('.select-avail-btn').forEach(btn => {
            btn.addEventListener('click', function(e) {
              e.stopPropagation();
              const row = this.closest('tr');
              selectAvailability(row);
            });
          });

          gridContainer.querySelectorAll('.avail-row').forEach(row => {
            row.addEventListener('click', function() {
              selectAvailability(this);
            });
          });

          function selectAvailability(row) {
            const date = row.dataset.date;
            const start = row.dataset.start;
            const end = row.dataset.end;

            document.getElementById('admin-shift-date').value = date;
            document.getElementById('admin-shift-start').value = start;
            document.getElementById('admin-shift-end').value = end;
            document.getElementById('admin-shift-date-display').textContent = date.split('-').reverse().join('/');
            document.getElementById('admin-shift-time-display').textContent = start?.substring(0,5) + ' - ' + end?.substring(0,5);
            document.getElementById('create-shift-btn').disabled = false;

            gridContainer.querySelectorAll('tr').forEach(r => r.style.background = '');
            row.style.background = '#cce5ff';

            checkVehicleConflict();
          }

        } catch (err) {
          gridContainer.innerHTML = '<p class="error-msg">Erro: ' + err.message + '</p>';
        }
      });

      function checkVehicleConflict() {
        const vehicleId = document.getElementById('admin-shift-vehicle').value;
        const date = document.getElementById('admin-shift-date').value;
        const start = document.getElementById('admin-shift-start').value;
        const end = document.getElementById('admin-shift-end').value;
        const warningDiv = document.getElementById('vehicle-conflict-warning');
        const createBtn = document.getElementById('create-shift-btn');

        if (!vehicleId || !date) {
          warningDiv.style.display = 'none';
          return;
        }

        const vehicle = allVehiclesData.find(v => v.id === parseInt(vehicleId));
        if (!vehicle) return;

        const conflict = allExistingShifts.find(s =>
          s.date === date &&
          s.start_time === start &&
          s.end_time === end &&
          s.vehicles?.license_plate === vehicle.license_plate
        );

        if (conflict) {
          warningDiv.innerHTML = `<p style="margin:0;color:#856404;">⚠️ Veículo <strong>${vehicle.license_plate}</strong> já atribuído a <strong>${conflict.profiles?.full_name || 'outro'}</strong> neste horário.</p>`;
          warningDiv.style.display = 'block';
          createBtn.disabled = true;
        } else {
          warningDiv.style.display = 'none';
          createBtn.disabled = false;
        }
      }

      document.getElementById('admin-shift-vehicle').addEventListener('change', checkVehicleConflict);

      document.getElementById('admin-shift-form').addEventListener('submit', async (e) => {
        e.preventDefault();
        const driverId = document.getElementById('admin-driver-select').value;
        const date = document.getElementById('admin-shift-date').value;
        const start = document.getElementById('admin-shift-start').value;
        const end = document.getElementById('admin-shift-end').value;
        const vehicleId = document.getElementById('admin-shift-vehicle').value;

        if (!driverId || !date || !start || !end || !vehicleId) {
          alert('Preencha todos os campos.');
          return;
        }

        const vehicle = allVehiclesData.find(v => v.id === parseInt(vehicleId));
        if (vehicle) {
          const conflict = allExistingShifts.find(s =>
            s.date === date && s.start_time === start && s.end_time === end &&
            s.vehicles?.license_plate === vehicle.license_plate
          );
          if (conflict) {
            alert(`Veículo ${vehicle.license_plate} já atribuído neste horário.`);
            return;
          }
        }

        try {
          const { error } = await supabase.from('shifts').insert({
            driver_id: driverId,
            vehicle_id: parseInt(vehicleId),
            date: date,
            start_time: start,
            end_time: end
          });
          if (error) {

      function compressImage(file) {
        return new Promise((resolve, reject) => {
          const reader = new FileReader();
          reader.readAsDataURL(file);
          reader.onload = (event) => {
            const img = new Image();
            img.src = event.target.result;
            img.onload = () => {
              const canvas = document.createElement('canvas');
              const maxWidth = 1200;
              let width = img.width, height = img.height;
              if (width > maxWidth) { height *= maxWidth / width; width = maxWidth; }
              canvas.width = width; canvas.height = height;
              canvas.getContext('2d').drawImage(img, 0, 0, width, height);
              canvas.toBlob(blob => {
                if (!blob) reject(new Error('Falha ao comprimir'));
                else resolve(new File([blob], file.name || 'foto.jpg', { type: 'image/jpeg' }));
              }, 'image/jpeg', 0.7);
            };
          };
          reader.onerror = reject;
        });
      }
    })();

      async function submitEndShift(shiftId) {
        try {
          const { data: currentShift, error: checkError } = await supabase
            .from('shifts')
            .select('status')
            .eq('id', shiftId)
            .single();

          if (checkError) throw checkError;
          if (currentShift.status !== 'started') {
            alert('Este turno já foi encerrado por outra ação. A lista será atualizada.');
            loadMyShifts();
            return;
          }

          const endKm = parseInt(document.getElementById(`end-km-${shiftId}`).value);
          if (!endKm || endKm < 0) {
            alert('Introduza um valor válido para os Km finais.');
            return;
          }

          const fuelLitres = document.getElementById(`fuel-litres-${shiftId}`).value || null;
          const fuelCost = document.getElementById(`fuel-cost-${shiftId}`).value || null;
          const invoiced = document.getElementById(`invoiced-${shiftId}`).value || null;
          const invoiceNum = document.getElementById(`invoice-num-${shiftId}`).value || null;
          const notes = document.getElementById(`notes-${shiftId}`).value || null;
          const photoFile = document.getElementById(`end-photo-${shiftId}`).files[0];
          let endPhotoUrl = null;

          if (photoFile) {
            const compressed = await compressImage(photoFile);
            const path = `uploads/${Date.now()}_${photoFile.name}`;
            const { error: uploadError } = await supabase.storage.from('fleet-photos').upload(path, compressed);
            if (uploadError) throw uploadError;
            const { data: urlData } = supabase.storage.from('fleet-photos').getPublicUrl(path);
            endPhotoUrl = urlData.publicUrl;
          }

          const updateData = {
            end_km: endKm,
            fuel_litres: fuelLitres ? parseFloat(fuelLitres) : null,
            fuel_cost: fuelCost ? parseFloat(fuelCost) : null,
            invoiced_amount: invoiced ? parseFloat(invoiced) : null,
            invoice_number: invoiceNum,
            notes: notes,
            closed: true,
            updated_at: new Date().toISOString()
          };
          if (endPhotoUrl) updateData.end_photo_url = endPhotoUrl;

          console.log('A encerrar turno ' + shiftId + '. Dados:', updateData);

          const { error: reportError } = await supabase
            .from('shift_reports')
            .update(updateData)
            .eq('shift_id', shiftId);

          if (reportError) {
            console.error('Erro ao atualizar relatório:', reportError);
            throw reportError;
          }

          console.log('Relatório atualizado. A atualizar shift para completed...');

          const { error: shiftError } = await supabase
            .from('shifts')
            .update({ status: 'completed' })
            .eq('id', shiftId)
            .eq('status', 'started');

          if (shiftError) {
            console.error('Erro ao atualizar shift:', shiftError);
            throw shiftError;
          }

          console.log('Shift atualizado para completed com sucesso!');

          alert('Turno encerrado com sucesso!');
          loadMyShifts();
          loadHistory();
        } catch (err) {
          console.error('Erro ao encerrar turno:', err);
          alert('Erro ao encerrar turno: ' + err.message);
          loadMyShifts();
        }
      }

      function compressImage(file) {
        return new Promise((resolve, reject) => {
          const reader = new FileReader();
          reader.readAsDataURL(file);
          reader.onload = (event) => {
            const img = new Image();
            img.src = event.target.result;
            img.onload = () => {
              const canvas = document.createElement('canvas');
              const maxWidth = 1200;
              let width = img.width;
              let height = img.height;
              if (width > maxWidth) {
                height = Math.round((height * maxWidth) / width);
                width = maxWidth;
              }
              canvas.width = width;
              canvas.height = height;
              const ctx = canvas.getContext('2d');
              ctx.drawImage(img, 0, 0, width, height);
              canvas.toBlob((blob) => {
                if (!blob) {
                  reject(new Error('Falha ao comprimir imagem'));
                  return;
                }
                const fileName = `${Date.now()}_${file.name || 'foto.jpg'}`;
                const compressedFile = new File([blob], fileName, { type: 'image/jpeg' });
                resolve(compressedFile);
              }, 'image/jpeg', 0.7);
            };
            img.onerror = () => reject(new Error('Erro ao carregar imagem'));
          };
          reader.onerror = () => reject(new Error('Erro ao ler ficheiro'));
        });
      }
    })();

// Compressão de imagem
function compressImage(file) {
  return new Promise((resolve, reject) => {
    const reader = new FileReader();
    reader.readAsDataURL(file);
    reader.onload = (event) => {
      const img = new Image();
      img.src = event.target.result;
      img.onload = () => {
        const canvas = document.createElement('canvas');
        const maxWidth = 1200;
        let width = img.width;
        let height = img.height;
        if (width > maxWidth) {
          height = Math.round((height * maxWidth) / width);
          width = maxWidth;
        }
        canvas.width = width;
        canvas.height = height;
        const ctx = canvas.getContext('2d');
        ctx.drawImage(img, 0, 0, width, height);
        canvas.toBlob(
          (blob) => {
            if (!blob) {
              reject(new Error('Falha ao comprimir imagem'));
              return;
            }
            const fileName = `${Date.now()}_${file.name || 'foto.jpg'}`;
            resolve(new File([blob], fileName, { type: 'image/jpeg' }));
          },
          'image/jpeg',
          0.7
        );
      };
      img.onerror = reject;
    };
    reader.onerror = reject;
  });
}

// Upload de foto
async function uploadPhoto(file) {
  if (!file) return null;
  try {
    const compressed = await compressImage(file);
    const filePath = `uploads/${compressed.name}`;
    const { data, error } = await window.supabase.storage
      .from('fleet-photos')
      .upload(filePath, compressed, {
        cacheControl: '3600',
        upsert: false,
      });
    if (error) throw error;
    const { data: urlData } = window.supabase.storage
      .from('fleet-photos')
      .getPublicUrl(filePath);
    return urlData.publicUrl;
  } catch (err) {
    console.error('Erro no upload:', err.message);
    alert('Erro ao enviar foto: ' + err.message);
    return null;
  }
}

async function callEdgeFunction(action, payload, sessionToken) {

    method: 'POST',
    headers: {

  });

}

const CACHE_NAME = 'frota-v3';
const urlsToCache = [
  '/',
  '/index.html',
  '/admin.html',
  '/driver.html',
  '/style.css',
  '/supabaseClient.js',
  '/manifest.json'
];

// Instalar o Service Worker e guardar em cache
self.addEventListener('install', event => {
  console.log('Service Worker a instalar...');
  event.waitUntil(
    caches.open(CACHE_NAME)
      .then(cache => {
        console.log('Cache aberto');
        return cache.addAll(urlsToCache);
      })
      .then(() => self.skipWaiting())
  );
});

// Ativar o Service Worker e limpar caches antigos
self.addEventListener('activate', event => {
  console.log('Service Worker ativado');
  event.waitUntil(
    caches.keys().then(cacheNames => {
      return Promise.all(
        cacheNames.filter(cacheName => cacheName !== CACHE_NAME)
          .map(cacheName => caches.delete(cacheName))
      );
    }).then(() => self.clients.claim())
  );
});

// Estratégia de cache: Network First, depois cache
self.addEventListener('fetch', event => {
  // Ignorar pedidos à API do Supabase
  if (event.request.url.includes('supabase.co')) {
    return;
  }

  event.respondWith(
    fetch(event.request)
      .then(response => {
        // Guardar em cache se a resposta for válida
        if (response && response.status === 200 && response.type === 'basic') {
          const responseToCache = response.clone();
          caches.open(CACHE_NAME).then(cache => {
            cache.put(event.request, responseToCache);
          });
        }
        return response;
      })
      .catch(() => {
        // Se a rede falhar, tentar servir da cache
        return caches.match(event.request);
      })
  );
});

# 🆘 Guia de Resolução de Problemas - Recursão RLS

## 🔴 Erro: "infinite recursion detected in policy for relation "profiles""

Este erro ocorre quando uma política RLS tenta referenciar a própria tabela, causando um loop infinito.

### Por que acontece?

A política RLS antigo fazia isto:

```sql
CREATE POLICY "profiles_select_admin" ON public.profiles
  FOR SELECT USING (
    EXISTS (SELECT 1 FROM public.profiles WHERE ...)  -- ❌ RECURSÃO!
  );
```

Quando o Supabase tenta verificar a política, ele tenta:
1. Ler dados da tabela `profiles`
2. A política diz: "primeiro verifica se é admin consultando `profiles`"
3. Isto causa outra consulta a `profiles`
4. Que dispara a política novamente... **infinito!**

---

## ✅ SOLUÇÃO RÁPIDA (5 minutos)

### Passo 1: Limpar o Database Completamente

1. Abra [Supabase Dashboard](https://supabase.com) → Seu Projeto → **SQL Editor**
2. Clique em **"New Query"**
3. **Cole este script de limpeza:**
   - Abra o ficheiro `database-cleanup.sql` (novo ficheiro)
   - Copie TODO o conteúdo
   - Cole no editor SQL do Supabase
4. Clique em **"Run"**
5. ✅ Tudo foi eliminado (tabelas, triggers, políticas, etc.)

### Passo 2: Executar o Script Novo (SEM RECURSÃO)

1. Abra o ficheiro **`database-simple-setup.sql`** no seu editor
2. Copie **TODO** o conteúdo (está corrigido agora - profiles SEM RLS)
3. Volte ao **Supabase SQL Editor** → Clique em **"New Query"**
4. Cole o conteúdo completo
5. Clique em **"Run"**

✅ **Pronto!** O database foi recriado SEM erros de recursão.

### Passo 3: Recriar o Utilizador Admin

1. Vá para **Authentication** → **Users**

    function compressImage(file) {
      return new Promise((resolve, reject) => {
        const reader = new FileReader();
        reader.readAsDataURL(file);
        reader.onload = (event) => {
          const img = new Image();
          img.src = event.target.result;
          img.onload = () => {
            const canvas = document.createElement('canvas');
            const maxWidth = 1200;
            let width = img.width, height = img.height;
            if (width > maxWidth) { height *= maxWidth / width; width = maxWidth; }
            canvas.width = width; canvas.height = height;
            canvas.getContext('2d').drawImage(img, 0, 0, width, height);
            canvas.toBlob(blob => {
              if (!blob) reject(new Error('Falha'));
              else resolve(new File([blob], file.name || 'foto.jpg', { type: 'image/jpeg' }));
            }, 'image/jpeg', 0.7);
          };
        };
        reader.onerror = reject;
      });
    }
